'use strict';
angular.module(moduleCore).directive('responsiveWidthGetter', function($window, Responsive) {
  return {
    restrict: 'A',
    link: function(scope, element, attrs) {
      var clear, onResize, resizeTimer, slugList;
      slugList = ['xs', 'sm', 'md', 'lg'];
      clear = function() {
        Responsive.xs = false;
        Responsive.sm = false;
        Responsive.md = false;
        return Responsive.lg = false;
      };
      onResize = function(apply) {
        var perform, style;
        style = element[0].currentStyle || $window.document.defaultView.getComputedStyle(element[0], '');
        clear();
        perform = function() {
          return Responsive[slugList[style['zIndex']]] = true;
        };
        if (apply) {
          return scope.$apply(perform);
        } else {
          return perform();
        }
      };
      resizeTimer = null;
      $window.addEventListener('resize', function() {
        if (resizeTimer !== null) {
          clearTimeout(resizeTimer);
        }
        return resizeTimer = setTimeout(function() {
          return onResize(true);
        }, 500);
      });
      return onResize(false);
    }
  };
}).provider('Responsive', function() {
  var el;
  el = document.createElement('div');
  el.setAttribute('responsive-width-getter', '');
  el.appendChild(document.createTextNode(''));
  document.body.appendChild(el);
  return {
    $get: function() {
      return {
        xs: false,
        sm: false,
        md: false,
        ls: false
      };
    }
  };
}).run(function($rootScope, Responsive) {
  return $rootScope.Responsive = Responsive;
});
